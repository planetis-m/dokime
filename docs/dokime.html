dokime — Compile-time validated SQL for Nimony.

import dokime
var db = openDatabase("mydb.sqlite")
let row = query(db, "SELECT id, name FROM users WHERE id = ?", 42'i64)
echo row.id     # int64
echo row.name   # string
  • template query(vanon: varargs): untyped

    Compile-time validated SQL query that returns exactly one row.

    The SQL string is checked against your development database at compile time. Set DOKIME_DATABASE_PATH to point to that database.

    let row = query(db, "SELECT id, name FROM users WHERE id = ?", userId)
echo row.id
  • template queryOne(vanon: varargs): untyped

    Alias for query.

  • template queryOpt(vanon: varargs): untyped

    Compile-time validated SQL query that may return zero rows.

    Returns Opt[tuple[...]]. Use when the absence of a matching row is expected and should not raise an error.

    let maybe = queryOpt(db, "SELECT name FROM users WHERE id = ?", uid)
if maybe.isSome:
  echo maybe.get.name
  • template rows(vanon: varargs): untyped

    Compile-time validated SQL query that returns all matching rows.

    Returns a value that can be iterated with for:

    for row in rows(db, "SELECT id, name FROM users"):
  echo row.id, " ", row.name
  • template exec(vanon: varargs): untyped

    Compile-time validated SQL command with no result columns.

    Use for INSERT, UPDATE, DELETE, DDL, BEGIN, COMMIT, and ROLLBACK. Returns the row count as int64.

    let affected = exec(db, "UPDATE users SET active = 1 WHERE id = ?", uid)
